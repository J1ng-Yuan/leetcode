class Solution {
public:
    string longestPalindrome(string s) {
            int n =s.length();
            string res="";
            for (int i=0;i<n;i++)
            {
                int k=0;
                while (i-k>=0 && i+k<n)
                {
                    if (s[i-k]==s[i+k])
                        k++;
                    else break;
                }
                if (2*k-1>res.length()) 
                    res=s.substr(i-k+1,2*k-1);
                int j=i;
                k=i+1;
                while (j>=0 && k<n)
                {
                    if (s[j]==s[k])
                        j--,k++;
                    else break;
                }
                if (k-j-1>res.length())
                    res=s.substr(j+1,k-j-1);
            }
            return res;
    }
};
